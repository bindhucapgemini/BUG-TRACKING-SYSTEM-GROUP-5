#include<stdio.h>
#include<string.h>
#include<stdlib.h>

 

#define TRUE 1
#define FALSE 0

 

int password_validation(char *);
int username_validation(char *);
int ticket_id_validation(int);
int project_id_validation(int);
int bug_id_validation(int);

 



int username_validation(char *username)
{
        int len,i;
        int small=0,caps=0,num=0,spec_char=0;
        len = strlen(username);

 

 

        if(len > 10)
        {
                return FALSE;
        }
        else
        {
                for(i=0;i<len;i++)
                {
                        if(username[i] >= 'a' && username[i] <= 'z')
                                small++;
                        else if(username[i] >= 'A' && username[i] <= 'Z')
                                caps++;
                        else if(username[i] >= '0' && username[i] <= '9')
                                num++;
                        else if(username[i] >= '!' && username[i] <= '/')
                                spec_char++;
                }
                if((len <= 10) || (num >= 1) || (spec_char >= 1))
                {
                        return TRUE;
                }
                else
                {
                        return FALSE;
                }
        }
}

 

int password_validation(char *password)
{
        int len,i;
        int small=0,caps=0,num=0,spec_char=0;
        len = strlen(password);
        if(len > 8)
        {
                return FALSE;
        }
        else
        {
                for(i=0;i<len;i++)
                {
                        if((password[i] >= 'a' && password[i] <= 'z'))
                                small++;
                        else if(password[i] >= 'A' && password[i] <= 'Z')
                                caps++;
                        else if(password[i] >= '0' && password[i] <= '9')
                                num++;
                        else if(password[i] == '!' || password[i] == '@' || password[i] == '#' ||password[i] == '$' || password[i] == '%' || password[i] == '^' || password[i] == '&' || password[i] == '*')
                                spec_char++;
                }
                if(small >= 1 && caps >= 1 && num >= 1 && spec_char >= 1)
                        return TRUE;
                else
                        return FALSE;
        }
}

 


int ticket_id_validation(int ticket_id)
{
        int count = 0;
        do
        {
                ticket_id /= 10;
                count++;
        }while(ticket_id != 0);

 

 

        if(count == 6)
                return TRUE;
        else
                return FALSE;
}
